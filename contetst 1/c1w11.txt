#include <iostream>
#include <cmath>

int main() {
    int A, B, C;
    std::cin >> A >> B >> C;

    int diffAB = std::abs(A - B);
    int diffAC = std::abs(A - C);

    if (diffAB < diffAC) {
        std::cout << "B " << diffAB << std::endl;
    } else {
        std::cout << "C " << diffAC << std::endl;
    }

    return 0;
}
